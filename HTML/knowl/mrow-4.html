\begin{align*} 1^3 + 2^3 + 3^3 + \cdots + k^3 +(k+1)^3 \amp = \frac{k^2(k+1)^2}{4} + (k+1)^3 \\ \amp = \frac{k^2(k+1)^2}{4} + \frac{4(k+1)^3}{4}\\ \amp = \frac{k^2(k+1)^2 + 4(k+1)^3}{4}\\ \amp = \frac{(k+1)^2 (k^2+4(k+1))}{4}\\ \amp = \frac{(k+1)^2 (k^2+4k+4)}{4}\\ \amp = \frac{(k+1)^2 (k+2)^2}{4}. \end{align*}
